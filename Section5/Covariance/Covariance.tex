Let $X$ be the standard normal random variable and $Y=X^{2}$.

\begin{question} 
     \begin{solution}
          a) Are $X$ and $Y$ independent ?
          \begin{multiple-choice}
               \choice{$\text{Yes}$}
                \choice[correct]{$\text{No}$}
          \end{multiple-choice} 
     \end{solution}
\end{question}

\begin{question}
     \begin{solution}
          b) What is the $\text{Cov}(X,Y)$ ?  \answer{$0$}.
     \end{solution}
\end{question}

Suppose that $X$, $Y$ and $Z$ are random variables with variances $1$, $2$ and $3$, respectively. Also, suppose that $\text{Cov}(X,Y)=4$, $\text{Cov}(X,Z)=-2$ and $\text{Cov}(Y,Z)=-1$.

\begin{question}
     \begin{solution}
     Determine $\text{Var}(X-Y+2Z) = $ \answer{$3$}.
     \end{solution}
\end{question}
