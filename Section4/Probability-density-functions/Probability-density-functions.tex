Suppose $X$ has a density $f(x)=c/x^{4}$ for $x>1$ and $f(x)=0$ otherwise, where $c$ is a constant.

\begin{question} 
     \begin{solution}
          a) Find $c = $  \answer{$3$}.
     \end{solution}
\end{question}

\begin{question}
     \begin{solution}
          b) Find $E(X) = $  \answer{$3/2$}.
     \end{solution}
\end{question}

\begin{question}
     \begin{solution}
          c) Find $\text{Var}(X) = $  \answer{$3/4$}.
     \end{solution}
\end{question}

Suppose $Y$ is a random variable whose density is $f(y)=d\,y\,(1-y)$ for $0 < y < 1$, and $f(y)=0$ otherwise, where $d$ is a constant.

\begin{question} 
     \begin{solution}
          a) Find $d = $  \answer{$6$}.
     \end{solution}
\end{question}

\begin{question}
     \begin{solution}
          b) Find $P(Y \leq 2/3) = $  \answer{$20/27$}.
     \end{solution}
\end{question}

\begin{question}
     \begin{solution}
          c) Find $P(Y\leq 1/4 ) = $  \answer{$5/32$}.
     \end{solution}
\end{question}

\begin{question}
     \begin{solution}
          d) Find $P(1/4 < Y \leq 2/3 ) = $  \answer{$20/27-5/32$}.
     \end{solution}
\end{question}

\begin{question}
     \begin{solution}
          e) Find $E(Y) = $  \answer{$1/2$}.
     \end{solution}
\end{question}

\begin{question}
     \begin{solution}
          e) Find $\text{Var}(Y) = $  \answer{$1/20$}.
     \end{solution}
\end{question}
